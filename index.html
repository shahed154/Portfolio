Multiplayer Interaction System (C++)
This interaction system allows players to interact with various objects in the game world. It uses raycasting to detect interactable objects and supports multiplayer replication. 
// Multiplayer Interaction System
void ABaseCharacter::Interact()
{
    // Prevent interaction while dead
    if (bIsDead) return;

    // Check if we are looking at an interactable
    if (CurrentInteractable != nullptr)
    {
        // Call the interact function on the interactable
        IInteractableInterface::Execute_OnInteract(CurrentInteractable, this);
        
        // Play interaction animation
        PlayAnimMontage(InteractionAnimMontage, 1.0f);
        
        // Replicate the interaction to all clients
        Server_Interact(CurrentInteractable);
    }
}

// Server implementation of interaction for multiplayer
void ABaseCharacter::Server_Interact_Implementation(AActor* Interactable)
{
    // Server validation
    if (Interactable && Interactable->Implements())
    {
        // Execute the interaction on the server
        IInteractableInterface::Execute_OnInteract(Interactable, this);
        
        // Tell all clients about the interaction
        Multi_PlayInteractAnim();
    }
}

Weapon System (C++ & Blueprint)
A comprehensive weapon system that handles weapon pickups, firing mechanics, and ammo management. The system supports multiple weapon types and is fully networked for multiplayer gameplay. 
// Weapon base class implementation
AWeaponBase::AWeaponBase()
{
    // Set this actor to call Tick() every frame
    PrimaryActorTick.bCanEverTick = true;
    
    // Create mesh component
    WeaponMesh = CreateDefaultSubobject(TEXT("WeaponMesh"));
    RootComponent = WeaponMesh;
    
    // Set network replication
    bReplicates = true;
    WeaponMesh->SetIsReplicated(true);
    
    // Default weapon properties
    MaxAmmo = 30;
    CurrentAmmo = MaxAmmo;
    FireRate = 0.1f;
    bCanFire = true;
    LastFireTime = 0.0f;
}

void AWeaponBase::Fire()
{
    // Check if we can fire
    if (!bCanFire || CurrentAmmo <= 0) return;
    
    // Get player controller
    AMyPlayerController* PC = Cast(GetOwner());
    if (!PC) return;
    
    // Get camera direction
    FVector CameraLocation;
    FRotator CameraRotation;
    PC->GetPlayerViewPoint(CameraLocation, CameraRotation);
    
    // Set trace parameters
    FVector TraceStart = CameraLocation;
    FVector TraceEnd = CameraLocation + (CameraRotation.Vector() * 10000.0f);
    
    // Perform trace
    FHitResult HitResult;
    FCollisionQueryParams QueryParams;
    QueryParams.AddIgnoredActor(this);
    QueryParams.AddIgnoredActor(GetOwner());
    
    // Process hit results
    if (GetWorld()->LineTraceSingleByChannel(HitResult, TraceStart, TraceEnd, ECC_Visibility, QueryParams))
    {
        // Apply damage if we hit a damageable actor
        AActor* HitActor = HitResult.GetActor();
        if (HitActor && HitActor->Implements())
        {
            IDamageableInterface::Execute_TakeDamage(HitActor, WeaponDamage, HitResult.ImpactPoint, PC);
        }
        
        // Spawn impact effects
        SpawnImpactEffects(HitResult);
    }
    
    // Spawn muzzle effects
    SpawnMuzzleEffects();
    
    // Consume ammo
    CurrentAmmo--;
    
    // Set fire timer
    bCanFire = false;
    LastFireTime = GetWorld()->GetTimeSeconds();
    GetWorldTimerManager().SetTimer(FireTimerHandle, this, &AWeaponBase::ResetFireTimer, FireRate, false);
    
    // Replicate fire event
    Server_Fire(TraceStart, TraceEnd);
}

API Integration with PokeAPI (JavaScript)
This code fetches random Pokemon and their moves from the PokeAPI. It showcases asynchronous JavaScript, API integration, and error handling. 
const BASE_URL = 'https://pokeapi.co/api/v2';

const getRandomPokemonId = () => Math.floor(Math.random() * 898) + 1;

async function getRandomPokemon() {
    try {
        const id = getRandomPokemonId();
        const response = await axios.get(`${BASE_URL}/pokemon/${id}`);
        return response.data;
    } catch (error) {
        console.error('Error getting random pokemon:', error);
        throw error;
    }
}

async function getRandomMove(pokemon, maxAttempts) {
    try {
        maxAttempts = 10;
        const allMoves = pokemon.moves;
        
        // Shuffle the moves array for randomness
        const shuffledMoves = [...allMoves].sort(() => 0.5 - Math.random())
        
        const attempts = shuffledMoves.length;
        
        for (let i = 0; i < attempts; i++) {
            const moveEntry = shuffledMoves[i]
            const moveUrl = moveEntry.move.url;
            
            const response = await axios.get(moveUrl);
            const moveData = response.data;
            
            // Only return moves that have power (for battle system)
            if (moveData.power !== null && moveData.power > 0) {
                return moveData;
            }
        }
        
        throw new Error(`No move with power found after ${attempts} attempts`);
        
    } catch (error) {
        console.error('Error getting random move with power', error);
        throw error;
    }
}

Battle System with Animation (JavaScript)
This battle system determines the winner based on move power and displays the battle results with timed animations using setTimeout. 
function battle(playerPokemon, playerMove, opponentPokemon, opponentMove) {
    const battleLog = [];

    battleLog.push(`${playerPokemon.name.toUpperCase()} VERSUS ${opponentPokemon.name.toUpperCase()}!`)
    battleLog.push(`${playerPokemon.name.toUpperCase()} used ${playerMove.name.toUpperCase()} (Power: ${playerMove.power})`);
    battleLog.push(`${opponentPokemon.name.toUpperCase()} used ${opponentMove.name.toUpperCase()} (Power: ${opponentMove.power})`);
    
    let winner;
    let winnerDetails;

    if (playerMove.power > opponentMove.power) {
        winner = 'player';
        winnerDetails = `${opponentPokemon.name.toUpperCase()} fainted! ${playerPokemon.name.toUpperCase()} with ${playerMove.name.toUpperCase()} Wins !  `;
    } else if (opponentMove.power > playerMove.power) {
        winner = 'opponent';
        winnerDetails = `${playerPokemon.name.toUpperCase()} fainted! ${opponentPokemon.name.toUpperCase()} with ${opponentMove.name.toUpperCase()} win! `;
    } else {
        winner = 'tie';
        winnerDetails = "Tied! THEY BOTH FAINTED";
    }

    battleLog.push(winnerDetails)

    return {
        winner: winner,
        battleLog: battleLog
    };
}

function displayBattleResults(battleLogElement, battleResults) {
    document.getElementById('roll-pokemon').disabled = true;
    document.getElementById('roll-move').disabled = true;
    document.getElementById('reset-game').disabled = true;

    battleLogElement.innerHTML = '
Battle Results
';
    
    // Display battle log entries with animation delay
    battleResults.battleLog.forEach((entry, index) => {
        setTimeout(() => {
            const logEntry = document.createElement('p');
            logEntry.textContent = entry;
            battleLogElement.appendChild(logEntry);
        }, index * 2500); 
    });

    // Re-enable reset button after battle animation finishes
    setTimeout(() => {
        document.getElementById('reset-game').disabled = false;
    }, (battleResults.battleLog.length * 2500));
}
